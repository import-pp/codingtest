# 계산하고 말하기

피보나치와 같은 방식으로 풀이함

```go
func countAndSay(n int) string {
    first := "1"
    var recur func (string, int) string
    
    if n == 1 {
        return "1"
    }
    if n == 2 {
        return "11"
    }
    
    recur = func (s string, term int) string {
        var result string
        var pre string
        var count int
        for _, v := range s {
            if pre == ""{
                pre = string(v)
            }
            
            if pre == string(v) {
                count++
            } else {
                result += strconv.Itoa(count) + string(pre)
                pre = string(v)
                count = 1
            }
        }
        
        result += strconv.Itoa(count) + string(pre)
        
        if n == term {
            return result
        }
        
        return recur(result, term + 1)
    }
    
    return recur(first, 2)
}
```
